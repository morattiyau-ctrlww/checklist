🎨 用 AI 畫 SVG & Canvas

不會寫 code 也沒關係 — 把需求丟給 AI，它就幫你畫好

🧠 怎麼玩？

你只需要做四件事：

  • 1. 想好你要畫什麼（一隻貓、一個圖表、一個 Logo…）
  • 2. 打開 Grok、Gemini、ChatGPT 或 Claude
  • 3. 把提示詞（Prompt）貼給它，它就幫你生出 HTML
  • 4. 把 AI 給你的 HTML 存成 .html 檔案，用瀏覽器打開就看得到了

👇 下面每一個例子，你都可以直接複製 prompt 去 AI 試試看。

煙火秀 (Canvas)

Canvas
🤖 複製這段 Prompt 給 AI 請寫出完整的 HTML 程式碼（包含 <canvas> + <script>），用 HTML5 Canvas 和 JavaScript 做煙火動畫。火箭從底部隨機位置發射升空，爆炸成彩色粒子並帶有拖尾效果，同時多發煙火在不同位置爆炸（紅、金、藍、綠、紫），深色夜空背景，煙火會慢慢淡出。尺寸 400x300。
📄 Canvas JavaScript 動畫 
JavaScriptconst canvas = document.getElementById('fw');
const ctx = canvas.getContext('2d');
const rockets = [], particles = [];
const colors = ['#f87171','#fbbf24','#60a5fa','#4ade80','#a78bfa','#f472b6'];

function launchRocket() {
  const x = 30 + Math.random() * (canvas.width - 60);
  rockets.push({
    x, y: canvas.height,
    targetY: 30 + Math.random() * (canvas.height * 0.45),
    speed: 3 + Math.random() * 3,
    angle: (Math.random() - 0.5) * 0.2,
    trail: [],
  });
}

function explode(r) {
  const color = colors[Math.floor(Math.random() * colors.length)];
  const count = 40 + Math.floor(Math.random() * 40);
  for (let i = 0; i < count; i++) {
    const a = Math.random() * Math.PI * 2;
    const sp = 1 + Math.random() * 4;
    particles.push({
      x: r.x, y: r.y,
      vx: Math.cos(a) * sp, vy: Math.sin(a) * sp,
      life: 1, decay: 0.008 + Math.random() * 0.012,
      size: 1.5 + Math.random() * 2.5,
      color,
    });
  }
}

function animate() {
  ctx.fillStyle = 'rgba(10,10,30,0.2)';
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  if (Math.random() < 0.06) launchRocket();
  if (Math.random() < 0.04) launchRocket();

  for (let i = rockets.length - 1; i >= 0; i--) {
    const r = rockets[i];
    r.x += Math.sin(r.angle) * 0.5;
    r.y -= r.speed;
    r.trail.push({x: r.x, y: r.y});
    if (r.trail.length > 8) r.trail.shift();
    if (r.y <= r.targetY) { explode(r); rockets.splice(i, 1); continue; }
  }

  for (const r of rockets) {
    for (let j = 1; j < r.trail.length; j++) {
      ctx.beginPath();
      ctx.moveTo(r.trail[j-1].x, r.trail[j-1].y);
      ctx.lineTo(r.trail[j].x, r.trail[j].y);
      ctx.strokeStyle = `rgba(255,255,255,${j / r.trail.length * 0.5})`;
      ctx.lineWidth = 1.5;
      ctx.stroke();
    }
    ctx.beginPath();
    ctx.arc(r.x, r.y, 2, 0, Math.PI * 2);
    ctx.fillStyle = '#fff';
    ctx.fill();
  }

  for (let i = particles.length - 1; i >= 0; i--) {
    const p = particles[i];
    p.x += p.vx; p.y += p.vy;
    p.vy += 0.04;
    p.life -= p.decay;
    if (p.life <= 0) { particles.splice(i, 1); continue; }
    ctx.globalAlpha = p.life;
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
    ctx.fillStyle = p.color;
    ctx.fill();
    ctx.globalAlpha = 1;
  }

  requestAnimationFrame(animate);
}
animate();

互動時鐘 (Canvas)

Canvas
🤖 複製這段 Prompt 給 AI 請寫出完整的 HTML 程式碼（包含 <canvas> + <script>），用 HTML5 Canvas 和 JavaScript 做一個類比時鐘。畫出圓形鐘面（1-12 小時標記）、即時更新的時針分針秒針，使用深色主題搭配發光霓虹風格的指針，秒針平滑轉動。尺寸 300x300。
00:00:00
📄 Canvas JavaScript 時鐘 
JavaScriptconst c = document.getElementById('clock');
const ctx = c.getContext('2d');
const cx = 150, cy = 150, r = 130;

function drawClock() {
  const now = new Date();
  const h = now.getHours() % 12, m = now.getMinutes(), s = now.getSeconds();

  ctx.clearRect(0, 0, 300, 300);
  // face
  ctx.beginPath();
  ctx.arc(cx, cy, r, 0, Math.PI * 2);
  ctx.fillStyle = '#0f172a';
  ctx.fill();
  ctx.strokeStyle = '#334155';
  ctx.lineWidth = 4;
  ctx.stroke();

  // hour markers ...
  for (let i = 0; i < 12; i++) {
    const a = (i * 30 - 90) * Math.PI / 180;
    ctx.beginPath();
    ctx.moveTo(cx + Math.cos(a)*(r-18), cy + Math.sin(a)*(r-18));
    ctx.lineTo(cx + Math.cos(a)*(r-6), cy + Math.sin(a)*(r-6));
    ctx.strokeStyle = i === 0 ? '#fbbf24' : '#64748b';
    ctx.lineWidth = i === 0 ? 4 : 2;
    ctx.stroke();
  }

  // hands ...
  // hour hand
  const ha = ((h + m/60) * 30 - 90) * Math.PI / 180;
  ctx.beginPath();
  ctx.moveTo(cx, cy);
  ctx.lineTo(cx + Math.cos(ha)*50, cy + Math.sin(ha)*50);
  ctx.strokeStyle = '#60a5fa';
  ctx.lineWidth = 6;
  ctx.stroke();
  // minute hand ...
}
setInterval(drawClock, 1000);
drawClock();

💡 給 AI 下 prompt 的小技巧

  • 要具體：不要只說「畫一隻貓」，而是說「用 SVG 畫一隻橘色的貓，圓臉、尖耳朵、長尾巴，大小 400x400」
  • 指定格式：開頭直接說 用 SVG 或 用 HTML5 Canvas + JavaScript
  • 給尺寸：加上 尺寸 500x400 避免 AI 亂猜大小
  • 要顏色：說清楚顏色，例如 深藍色背景 + 金色文字
  • 要動畫：如果是 Canvas，說 加上動畫，粒子會慢慢往上飄
  • 一鍵複製：直接跟 AI 說 給我一份完整的 HTML 檔案，打開就能用

🤖 推薦 AI 工具

  • Grok (xAI) — 免費，擅長生成 SVG，回覆直接

  • ChatGPT (OpenAI) — 穩定，適合複雜動畫
